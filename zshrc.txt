loghistory() {
    if [[ -f "$HISTFILE" ]]; then
        perl -lne 'm#: (\d+):\d+;(.+)# && printf "%s :: %s\n", scalar localtime $1, $2' "$HISTFILE" | less
    else
        echo "Error: History file not found at $HISTFILE"
    fi
}
